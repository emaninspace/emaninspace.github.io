stuff <- mydata_clean %>%
  distinct(SERIALNO, PINCP, .keep_all = TRUE) %>%
  filter(!is.na(PINCP), 
         !is.na(JWTR_new),
         PINCP >= 0,
         PINCP <= 250000) %>%
  mutate(JWTR_new = recode(JWTR_new,
                           "Bus or trolley bus" = "Other",
                           "Streetcar" = "Other",
                           "Subway" = "Other",
                           "Taxicab" = "Other",
                           "Motorcycle" = "Other",
                           "Ferryboat" = "Other",
                           "Railroad" = "Other",
                           "Worked at home" = "Other",
                           "Other method" = "Other"
                         )) %>%
  mutate(PINCP_binned = cut(PINCP,
                            breaks=seq(from=0,
                                       to=250000,
                                       by=50000),
                            dig.lab=7,
                            right = FALSE
                           )) %>%
  filter(!is.na(PINCP_binned))

stuff %>%
  tabyl(PINCP_binned, JWTR_new) %>%
  adorn_percentages("row") %>%
  adorn_pct_formatting(digits = 0) %>%
  kable() %>%
  kable_styling(full_width = F, position="center")
